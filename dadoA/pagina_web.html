SumaDeCuatroDados <- function(n = 100) {
  tiradas.CuatroDados = CuatroDados(n)

  #resultado = sapply(1:n, sum)
  resultado = vector(mode="numeric", n)
  for (i in 1:n) {
    resultado[i] = tiradas.CuatroDados[[1]][i] + tiradas.CuatroDados[[2]][i] + tiradas.CuatroDados[[3]][i] + tiradas.CuatroDados[[4]][i]
  }

  return(resultado)
}

  sumas = SumaDeCuatroDados(n)
